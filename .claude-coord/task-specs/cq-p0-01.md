## Implementation Details

```python
import ipaddress
import socket
import urllib.parse

BLOCKED_HOSTS = [
    "localhost", "127.0.0.1", "0.0.0.0",
    "169.254.169.254",  # AWS/Azure metadata
    "metadata.google.internal",  # GCP metadata
    "::1", "::ffff:127.0.0.1",  # IPv6 localhost
]

BLOCKED_NETWORKS = [
    ipaddress.IPv4Network("10.0.0.0/8"),
    ipaddress.IPv4Network("172.16.0.0/12"),
    ipaddress.IPv4Network("192.168.0.0/16"),
    ipaddress.IPv4Network("127.0.0.0/8"),
    ipaddress.IPv6Network("::1/128"),
    ipaddress.IPv6Network("fe80::/10"),  # Link-local
]

def validate_url_safety(url: str) -> Tuple[bool, Optional[str]]:
    """Validate URL doesn't target internal resources."""
    try:
        parsed = urllib.parse.urlparse(url)
        hostname = parsed.hostname
        
        if not hostname:
            return False, "Invalid URL: missing hostname"
        
        # Block known dangerous hostnames
        if hostname.lower() in BLOCKED_HOSTS:
            return False, f"Access to {hostname} is forbidden"
        
        # Resolve and check IP address
        try:
            ip = ipaddress.ip_address(socket.gethostbyname(hostname))
            for network in BLOCKED_NETWORKS:
                if ip in network:
                    return False, f"Access to private network is forbidden"
        except (socket.gaierror, ValueError) as e:
            return False, f"Cannot resolve hostname: {e}"
            
        return True, None
    except Exception as e:
        return False, f"Invalid URL: {e}"
```

def test_ssrf_blocks_localhost():
    scraper = WebScraper()
    result = scraper.execute(url="http://localhost:6379")
    assert not result.success
    assert "forbidden" in result.error.lower()

def test_ssrf_blocks_private_ips():
    dangerous_urls = [
        "http://10.0.0.1/admin",
        "http://192.168.1.1/config",
        "http://172.16.0.1/secrets",
    ]
    scraper = WebScraper()
    for url in dangerous_urls:
        result = scraper.execute(url=url)
        assert not result.success

def test_ssrf_blocks_cloud_metadata():
    scraper = WebScraper()
    result = scraper.execute(url="http://169.254.169.254/latest/meta-data")
    assert not result.success
