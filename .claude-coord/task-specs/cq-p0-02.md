# src/utils/secrets.py
import os
import re
from typing import Optional, Dict, Any
from cryptography.fernet import Fernet

class SecretReference:
    """Reference to a secret in external store."""
    
    PATTERNS = {
        'env': re.compile(r'\$\{env:([A-Z_]+)\}'),
        'vault': re.compile(r'\$\{vault:([a-z0-9/_-]+)\}'),
        'aws': re.compile(r'\$\{aws:([a-z0-9/_-]+)\}'),
    }
    
    @classmethod
    def resolve(cls, reference: str) -> str:
        """Resolve secret reference to actual value."""
        for provider, pattern in cls.PATTERNS.items():
            match = pattern.match(reference)
            if match:
                return cls._resolve_provider(provider, match.group(1))
        
        # If not a reference, assume literal (for backward compat)
        return reference
    
    @classmethod
    def _resolve_provider(cls, provider: str, key: str) -> str:
        if provider == 'env':
            value = os.environ.get(key)
            if value is None:
                raise ValueError(f"Environment variable {key} not set")
            return value
        elif provider == 'vault':
            # TODO: Implement Vault integration
            raise NotImplementedError("Vault provider not yet implemented")
        elif provider == 'aws':
            # TODO: Implement AWS Secrets Manager
            raise NotImplementedError("AWS Secrets Manager not yet implemented")

class SecureCredential:
    """Encrypted credential storage in memory."""
    
    def __init__(self, value: str):
        self._key = Fernet.generate_key()
        self._cipher = Fernet(self._key)
        self._encrypted = self._cipher.encrypt(value.encode())
    
    def get(self) -> str:
        """Decrypt and return credential."""
        return self._cipher.decrypt(self._encrypted).decode()
    
    def __repr__(self):
        return "SecureCredential(***REDACTED***)"
    
    def __str__(self):
        return "***REDACTED***"

class InferenceConfig(BaseModel):
    # DEPRECATED: api_key field removed
    # Use api_key_ref instead
    api_key_ref: str = Field(
        default="${env:OPENAI_API_KEY}",
        description="Secret reference: ${env:VAR_NAME}, ${vault:path}, ${aws:secret-id}"
    )
    
    @model_validator(mode='after')
    def warn_deprecated_api_key(self):
        # Check if old config has api_key field (backward compat)
        if hasattr(self, 'api_key') and self.api_key:
            import warnings
            warnings.warn(
                "api_key field is deprecated. Use api_key_ref with ${env:VAR_NAME}",
                DeprecationWarning
            )
            self.api_key_ref = self.api_key
        return self

def test_env_secret_resolution():
    os.environ['TEST_API_KEY'] = 'sk-test123'
    ref = SecretReference.resolve('${env:TEST_API_KEY}')
    assert ref == 'sk-test123'

def test_missing_env_raises_error():
    with pytest.raises(ValueError, match="not set"):
        SecretReference.resolve('${env:NONEXISTENT_KEY}')

def test_secrets_redacted_in_logs(caplog):
    os.environ['API_KEY'] = 'sk-secret'
    config = load_config('agent.yaml')
