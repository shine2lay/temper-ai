### Class Structure

```python
from abc import ABC
from typing import Dict, Any
from src.compiler.execution_engine import (
    ExecutionEngine,
    CompiledWorkflow,
    ExecutionMode
)
from src.compiler.langgraph_compiler import (
    LangGraphCompiler,
    WorkflowExecutor,
    WorkflowState
)


class LangGraphCompiledWorkflow(CompiledWorkflow):
    """LangGraph-specific compiled workflow wrapper.

    Wraps a LangGraph StateGraph to present ExecutionEngine interface.
    """

    def __init__(
        self,
        graph: Any,  # Compiled LangGraph StateGraph
        workflow_config: Dict[str, Any],
        tracker: Any = None
    ):
        """Initialize compiled workflow.

        Args:
            graph: Compiled LangGraph StateGraph
            workflow_config: Original workflow configuration
            tracker: Optional ExecutionTracker
        """
        self.graph = graph
        self.workflow_config = workflow_config
        self.tracker = tracker

    def invoke(self, state: Dict[str, Any]) -> Dict[str, Any]:
        """Execute workflow synchronously."""
        # Prepare state with tracker if available
        workflow_state = WorkflowState(**state)
        if self.tracker:
            workflow_state["tracker"] = self.tracker

        # Execute graph
        result = self.graph.invoke(workflow_state)
        return result

    async def ainvoke(self, state: Dict[str, Any]) -> Dict[str, Any]:
        """Execute workflow asynchronously."""
        workflow_state = WorkflowState(**state)
        if self.tracker:
            workflow_state["tracker"] = self.tracker

        result = await self.graph.ainvoke(workflow_state)
        return result

    def get_metadata(self) -> Dict[str, Any]:
        """Get workflow metadata."""
        workflow = self.workflow_config.get("workflow", self.workflow_config)
        stages = workflow.get("stages", [])

        return {
            "engine": "langgraph",
            "version": "0.2.0",  # LangGraph version
            "config": self.workflow_config,
            "stages": [
                stage if isinstance(stage, str)
                else stage.get("name", stage.get("stage_name", str(stage)))
                for stage in stages
            ]
        }

    def visualize(self) -> str:
        """Generate Mermaid diagram of workflow.

        Returns:
            Mermaid flowchart string
        """
        workflow = self.workflow_config.get("workflow", self.workflow_config)
        stages = workflow.get("stages", [])
        stage_names = [
            stage if isinstance(stage, str)
            else stage.get("name", stage.get("stage_name", str(stage)))
            for stage in stages
        ]

        # Generate Mermaid flowchart
        lines = ["flowchart TD"]
        lines.append("    START([Start])")

        for i, stage in enumerate(stage_names):
            lines.append(f"    {stage}[{stage}]")

        lines.append("    END([End])")

        # Add edges
        lines.append("    START --> " + stage_names[0])
        for i in range(len(stage_names) - 1):
            lines.append(f"    {stage_names[i]} --> {stage_names[i+1]}")
        lines.append(f"    {stage_names[-1]} --> END")

        return "\n".join(lines)


class LangGraphExecutionEngine(ExecutionEngine):
    """LangGraph execution engine adapter.

    Wraps LangGraphCompiler to implement ExecutionEngine interface.
    Preserves all existing M2 functionality while enabling future
    engine swapping.
    """

    def __init__(self, tool_registry=None, config_loader=None):
        """Initialize engine.

        Args:
            tool_registry: ToolRegistry instance
            config_loader: ConfigLoader instance
        """
        self.compiler = LangGraphCompiler(
            tool_registry=tool_registry,
            config_loader=config_loader
        )
        self.tool_registry = tool_registry
        self.config_loader = config_loader

    def compile(self, workflow_config: Dict[str, Any]) -> CompiledWorkflow:
        """Compile workflow configuration.

        Args:
            workflow_config: Framework-agnostic workflow config

        Returns:
            LangGraphCompiledWorkflow instance

        Raises:
            ValueError: If workflow config is invalid
        """
        # Use existing LangGraphCompiler
        graph = self.compiler.compile(workflow_config)

        # Wrap in CompiledWorkflow interface
        return LangGraphCompiledWorkflow(
            graph=graph,
            workflow_config=workflow_config,
            tracker=None  # Tracker passed at execution time
        )

    def execute(
        self,
        compiled_workflow: CompiledWorkflow,
        input_data: Dict[str, Any],
        mode: ExecutionMode = ExecutionMode.SYNC
    ) -> Dict[str, Any]:
        """Execute compiled workflow.

        Args:
            compiled_workflow: LangGraphCompiledWorkflow instance
            input_data: Input data dict
            mode: Execution mode (SYNC or ASYNC)

        Returns:
            Final workflow state with stage outputs

        Raises:
            TypeError: If compiled_workflow is not LangGraphCompiledWorkflow
            NotImplementedError: If mode is STREAM (not supported in M2)
        """
        # Type check
        if not isinstance(compiled_workflow, LangGraphCompiledWorkflow):
            raise TypeError(
                f"Expected LangGraphCompiledWorkflow, got {type(compiled_workflow)}"
            )

        # Mode check
        if mode == ExecutionMode.STREAM:
            raise NotImplementedError("STREAM mode not supported in M2")

        if mode == ExecutionMode.ASYNC:
            # For ASYNC, we need to call ainvoke (which is async)
            # Since this method is sync, we wrap it
            import asyncio
            return asyncio.run(compiled_workflow.ainvoke(input_data))

        # SYNC mode (default)
        return compiled_workflow.invoke(input_data)

    def supports_feature(self, feature: str) -> bool:
        """Check if engine supports feature.

        Args:
            feature: Feature name

        Returns:
            True if supported, False otherwise
        """
        # M2 LangGraph capabilities
        supported = {
            "sequential_stages",      # ✓ Sequential execution
            "parallel_stages",        # ✓ LangGraph supports parallel branches
            "conditional_routing",    # ✓ Conditional edges
            "checkpointing",          # ✓ LangGraph memory/checkpointing
            "state_persistence",      # ✓ State passed between nodes
        }

        # M3+ features not yet implemented
        unsupported = {
            "convergence_detection",   # ✗ Requires custom logic
            "dynamic_stage_injection", # ✗ Requires graph modification
            "nested_workflows",        # ✗ Not implemented in M2
        }

        return feature in supported
```

## Notes

**Why Adapter Pattern:**
- Preserves existing LangGraphCompiler without modification
- Composition over inheritance (more flexible)
- Minimal refactoring risk (existing code untouched)
- Easy to test adapter in isolation

**Critical:**
- Do NOT modify LangGraphCompiler - wrap it
- All existing tests must still pass
- No performance regression
- Preserve all M2 functionality (tracking, tools, config loading)

**Future:**
- M3+ will need to update supports_feature() as new features added
- STREAM mode implementation deferred to M4
- Nested workflows support deferred to M5
