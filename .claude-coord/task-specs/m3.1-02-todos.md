```python
def _load_tools(self, tool_names: List[str]) -> List[BaseTool]:
    """Load and initialize tools for agent"""
    tools = []
    for tool_name in tool_names:
        tool = self.tool_registry.get_tool(tool_name)
        if tool is None:
            logger.warning(f"Tool {tool_name} not found, skipping")
            continue

        # Initialize tool with agent context
        tool.set_context(self.agent_id, self.workflow_id)
        tools.append(tool)

    logger.info(f"Loaded {len(tools)} tools for agent {self.agent_id}")
    return tools
```

**Solution:**
```python
class Service(ABC):
    def __init__(self, tracker: Optional[ExecutionTracker] = None):
        self.tracker = tracker
        self.service_id = str(uuid.uuid4())
        self.start_time: Optional[datetime] = None

    def start(self):
        """Start service with observability tracking"""
        self.start_time = datetime.now()

        if self.tracker:
            self.tracker.track_event(
                "service_started",
                {
                    "service_id": self.service_id,
                    "service_name": self.__class__.__name__,
                    "start_time": self.start_time.isoformat()
                }
            )

        logger.info(f"Service {self.__class__.__name__} started")

    def stop(self):
        """Stop service with observability tracking"""
        if self.tracker and self.start_time:
            duration = (datetime.now() - self.start_time).total_seconds()
            self.tracker.track_event(
                "service_stopped",
                {
                    "service_id": self.service_id,
                    "service_name": self.__class__.__name__,
                    "duration_seconds": duration
                }
            )

        logger.info(f"Service {self.__class__.__name__} stopped")
```

**Test:**
```python
def test_service_observability_tracking():
    tracker = Mock(spec=ExecutionTracker)

    class TestService(Service):
        pass

    service = TestService(tracker=tracker)
    service.start()

    # Verify start tracked
    tracker.track_event.assert_called_once()
    call_args = tracker.track_event.call_args[0]
    assert call_args[0] == "service_started"

    service.stop()

    # Verify stop tracked with duration
    assert tracker.track_event.call_count == 2
    stop_call = tracker.track_event.call_args_list[1][0]
    assert stop_call[0] == "service_stopped"
    assert "duration_seconds" in stop_call[1]
```
