```python
class Experiment(SQLModel, table=True):
    """A/B test experiment definition."""
    id: str  # UUID primary key
    name: str  # Unique experiment name
    description: str
    status: Literal["draft", "running", "paused", "stopped", "completed"]

    # Assignment
    assignment_strategy: str  # "random", "hash", "stratified"
    traffic_allocation: Dict[str, float]  # {"control": 0.5, "variant_a": 0.5}

    # Metrics
    primary_metric: str
    secondary_metrics: List[str]
    guardrail_metrics: List[Dict[str, Any]]  # [{"metric": "error_rate", "max_value": 0.05}]

    # Statistics
    confidence_level: float = 0.95
    min_sample_size_per_variant: int = 100

    # Results
    winner_variant_id: Optional[str]
    total_executions: int = 0

    # Timestamps
    created_at: datetime
    started_at: Optional[datetime]
    stopped_at: Optional[datetime]


class Variant(SQLModel, table=True):
    """Variant configuration within experiment."""
    id: str  # UUID primary key
    experiment_id: str  # Foreign key
    name: str  # "control", "variant_a"
    description: str
    is_control: bool = False

    # Configuration
    config_type: Literal["agent", "stage", "workflow", "prompt"]
    config_overrides: Dict[str, Any]

    # Traffic
    allocated_traffic: float
    actual_traffic: float = 0.0

    # Metrics (cached)
    total_executions: int = 0
    successful_executions: int = 0
    failed_executions: int = 0


class VariantAssignment(SQLModel, table=True):
    """Assignment of workflow to variant."""
    id: str  # UUID primary key
    experiment_id: str  # Foreign key
    variant_id: str  # Foreign key
    workflow_execution_id: str  # Foreign key to WorkflowExecution (unique)

    # Assignment metadata
    assigned_at: datetime
    assignment_strategy: str
    assignment_context: Optional[Dict[str, Any]]

    # Execution tracking
    execution_status: Literal["pending", "running", "completed", "failed"]
    execution_started_at: Optional[datetime]
    execution_completed_at: Optional[datetime]

    # Metrics (denormalized for performance)
    metrics: Optional[Dict[str, float]]


class ExperimentResult(SQLModel, table=True):
    """Statistical analysis results."""
    id: str  # UUID primary key
    experiment_id: str  # Foreign key
    analyzed_at: datetime
    sample_size: int

    # Variant metrics
    variant_metrics: Dict[str, Dict[str, Any]]
    # {"control": {"mean": 45.2, "std": 5.1, "median": 44.0, "p95": 52.3, "count": 150}}

    # Statistical tests
    statistical_tests: Dict[str, Dict[str, Any]]
    # {"control_vs_variant_a": {"test": "t-test", "p_value": 0.003, "significant": true}}

    # Recommendations
    recommendation: Literal["continue", "stop_winner", "stop_no_difference", "stop_guardrail_violation"]
    recommended_winner: Optional[str]
    confidence: float
```

class AssignmentStrategy(ABC):
    @abstractmethod
    def assign(
        self,
        experiment: Experiment,
        variants: List[Variant],
        execution_id: str,
        context: Optional[Dict[str, Any]]
    ) -> str:
        """Return variant_id."""


class RandomAssignment(AssignmentStrategy):
    def assign(self, ...) -> str:
        # Weighted random selection based on traffic_allocation
        return random.choices(variant_ids, weights=traffics)[0]


class HashAssignment(AssignmentStrategy):
    def assign(self, ...) -> str:
        # Deterministic: hash(execution_id) % total → variant
        hash_val = int(hashlib.md5(execution_id.encode()).hexdigest(), 16)
        # Map hash to variant based on traffic allocation
        return self._hash_to_variant(hash_val, variants)

### Configuration Deep Merge

```python
class ConfigManager:
    def merge_config(
        self,
        base_config: Dict[str, Any],
        variant_overrides: Dict[str, Any]
    ) -> Dict[str, Any]:
        """Deep merge variant overrides into base config."""
        merged = copy.deepcopy(base_config)
        self._deep_merge(merged, variant_overrides)
        return merged

    def _deep_merge(self, base: Dict, overrides: Dict) -> None:
        """Recursive deep merge."""
        for key, value in overrides.items():
            if key in base and isinstance(base[key], dict) and isinstance(value, dict):
                self._deep_merge(base[key], value)
            else:
                base[key] = value
```

class StatisticalAnalyzer:
    def analyze_experiment(self, experiment_id: str) -> ExperimentResult:
        # 1. Load all assignments and metrics
        assignments = self._load_assignments(experiment_id)

        # 2. Group by variant, calculate aggregates
        variant_metrics = self._calculate_aggregates(assignments)

        # 3. Run t-tests between control and each treatment
        statistical_tests = self._run_hypothesis_tests(variant_metrics)

        # 4. Check guardrails
        guardrail_violations = self._check_guardrails(variant_metrics)

        # 5. Generate recommendation
        recommendation = self._generate_recommendation(
            statistical_tests,
            guardrail_violations
        )

        return ExperimentResult(...)

    def _run_t_test(
        self,
        control_values: List[float],
        treatment_values: List[float]
    ) -> Dict[str, Any]:
        from scipy.stats import ttest_ind

        t_stat, p_value = ttest_ind(control_values, treatment_values)

        return {
            "test": "t-test",
            "t_statistic": t_stat,
            "p_value": p_value,
            "significant": p_value < (1 - self.confidence_level),
            "improvement": (mean(treatment) - mean(control)) / mean(control)
        }
